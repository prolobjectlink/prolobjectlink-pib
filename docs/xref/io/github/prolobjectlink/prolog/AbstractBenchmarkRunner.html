22  package io.github.prolobjectlink.prolog;

24  import static java.util.logging.Level.SEVERE;
25  
26  import java.io.File;
27  import java.io.FileNotFoundException;
28  import java.io.IOException;
29  import java.io.PrintWriter;
30  import java.security.CodeSource;
31  import java.security.ProtectionDomain;
32  import java.util.Collection;
33  import java.util.logging.Logger;
34  
35  import org.openjdk.jmh.annotations.Mode;
36  import org.openjdk.jmh.results.RunResult;
37  
38  public abstract class AbstractBenchmarkRunner implements BenchmarkRunner {
39  
40  	public final String getCurrentPath() {
41  		Class<?> c = AbstractBenchmarkRunner.class;
42  		ProtectionDomain d = c.getProtectionDomain();
43  		CodeSource s = d.getCodeSource();
44  		return s.getLocation().getPath();
45  	}
46  
47  	public final void output(Collection<RunResult> results) {
48  
49  		File out = null;
50  		File outfile = null;
51  		PrintWriter writer = null;
52  		String name = getClass().getSimpleName();
53  		File jar = new File(getCurrentPath());
54  		File lib = jar.getParentFile();
55  		File dist = lib.getParentFile();
56  
57  		try {
58  
59  			out = new File(dist.getCanonicalPath() + File.separator + "out");
60  			out.mkdirs();
61  			outfile = new File(out.getCanonicalPath() + File.separator + name + ".txt");
62  
63  			if (outfile.createNewFile()) {
64  				writer = new PrintWriter(outfile);
65  
66  				for (RunResult result : results) {
67  
68  					Mode mode = result.getParams().getMode();
69  
70  					double score = result.getPrimaryResult().getScore();
71  					double scoerr = result.getPrimaryResult().getScoreError();
72  					double count = result.getPrimaryResult().getSampleCount();
73  					String label = result.getPrimaryResult().getLabel();
74  					String unit = result.getPrimaryResult().getScoreUnit();
75  
76  					double min = result.getPrimaryResult().getStatistics().getMin();
77  					double max = result.getPrimaryResult().getStatistics().getMax();
78  					double stdev = result.getPrimaryResult().getStatistics().getStandardDeviation();
79  
80  					writer.print(name + "." + label);
81  					writer.print("\t");
82  					writer.print(mode);
83  					writer.print("\t");
84  					writer.print(count);
85  					writer.print("\t");
86  					writer.print(min);
87  					writer.print("\t");
88  					writer.print(score);
89  					writer.print("\t");
90  					writer.print(max);
91  					writer.print("\t");
92  					writer.print(scoerr);
93  					writer.print("\t");
94  					writer.print(stdev);
95  					writer.print("\t");
96  					writer.println(unit);
97  
98  				}
99  
100 			}
101 
102 		} catch (FileNotFoundException e) {
103 			Logger.getLogger(getClass().getName()).log(SEVERE, null, e);
104 		} catch (IOException e) {
105 			Logger.getLogger(getClass().getName()).log(SEVERE, null, e);
106 		} finally {
107 			if (writer != null) {
108 				writer.close();
109 			}
110 		}
111 
112 	}

114 }
